using System;

class HomeTask1 {
  public static void Main() {
    Console.WriteLine(MaxArr(10,50,60,-70,1));
  }
  
  static int MaxArr(params int[] arr){
      int max = arr[0];
      for(int i = 1 ; i < arr.Length ; i++){
          if(arr[i] > max){
                max = arr[i];
          }
      }
      return max;
  }
  
}
